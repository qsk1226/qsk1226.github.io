##mysql 常用命令
mysql --help | grep my.cnf 此命令会告诉我们会在哪些位置查找配置文件

mysql -h host -u user -p pwd -P port

mysql的缓存推荐关闭

# 设置缓存大小
> set global query_cache_size=4000;

#查看缓存是否开启，必须配置文件中进行配置
> show variables like '%query_cache_type%' 

#查看当前使用的MySql数据库锁支持的存储引擎。也可以通过查找information_schema架构下的ENGINES表
> show engines;

# 查看当前数据库默认的存储引擎
>show variables LIKE '%storage_engine%';

# 查看数据文件存储的位置
> show variables like 'datadir';



#观察innodb的过去某个时间范围内的状态
> show engine innodb status 

#查看缓冲池大小
> show variables like 'innodb_buffer_pool_size'  

# 查看有几个缓冲池实例
> show variables like 'innodb_buffer_pool_instances' 

#设置多长时间之后热点页被放入LRU列表前端
> set global innodb_old_block_time=1000; 

#查看重做日志缓冲池大小
> show variables like 'innodb_log_buffer_size' 
# 了解各种sql的执行频率
> show status 
> show global status
> show status like 'Com_%'  
> insert into select * from table2 where column=100;

# Com_XXX表示每个XXX语句执行的次数，select、update、insert、delete
# 只针对 innoDB 引擎的参数：
# InnoDB_rows_read 执行 select 操作的次数
# InnoDB_rows_updated 执行 update 操作的次数
# InnoDB_rows_inserted 执行 insert 操作的次数
# InnoDB_rows_deleted 执行 delete 操作的次数
